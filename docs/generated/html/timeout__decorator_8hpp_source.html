16struct TimeoutConfig {
17 std::chrono::milliseconds timeout_duration{5000};
18 bool throw_on_timeout{true};
19 bool enable_progress_callback{false};
20 std::chrono::milliseconds progress_interval{100};
21 std::function<bool(double, double, std::chrono::milliseconds)> progress_callback;
22
23 // Validation settings
24 bool validate_timeout_duration{true};
25 std::chrono::milliseconds min_timeout_duration{10}; // Minimum 10ms
26 std::chrono::milliseconds max_timeout_duration{std::chrono::hours{24}}; // Maximum 24h
27
32 void validate() const {
33 if (validate_timeout_duration) {
34 if (timeout_duration < min_timeout_duration) {
35 throw std::invalid_argument(
36 "Timeout duration " + std::to_string(timeout_duration.count()) +
37 "ms is below minimum " + std::to_string(min_timeout_duration.count()) + "ms");
38 }
39 if (timeout_duration > max_timeout_duration) {
40 throw std::invalid_argument(
41 "Timeout duration " + std::to_string(timeout_duration.count()) +
42 "ms exceeds maximum " + std::to_string(max_timeout_duration.count()) + "ms");
43 }
44 }
45
46 if (enable_progress_callback && !progress_callback) {
47 throw std::invalid_argument("Progress callback enabled but no callback provided");
48 }
49
50 if (progress_interval <= std::chrono::milliseconds{0}) {
51 throw std::invalid_argument("Progress interval must be positive");
52 }
53 }

59struct TimeoutResult {
60 bool completed{false};
61 std::chrono::milliseconds elapsed_time{0};
62 double final_time{0.0};
63 std::string error_message;
64 bool user_cancelled{false};
65 size_t progress_callbacks_made{0};
66
67 bool is_success() const { return completed && error_message.empty() && !user_cancelled; }
68 bool is_timeout() const { return !completed && error_message.find("timeout") != std::string::npos; }
69 bool is_user_cancelled() const { return user_cancelled; }
70 bool is_error() const { return !completed && !error_message.empty() && !is_timeout(); }
71
75 std::string status_description() const {
76 if (is_success()) return "Integration completed successfully";
77 if (is_timeout()) return "Integration timed out";
78 if (is_user_cancelled()) return "Integration cancelled by user";
79 if (is_error()) return "Integration failed: " + error_message;
80 return "Integration status unknown";
81 }
82};

124 TimeoutResult integrate_with_timeout(typename IntegratorDecorator<S, T>::state_type& state,
125 T dt, T end_time) {
126 const auto start_time = std::chrono::high_resolution_clock::now();
127 TimeoutResult result;
128 result.final_time = this->current_time();
129
130 try {
131 if (config_.enable_progress_callback && config_.progress_callback) {
132 result = integrate_with_progress_monitoring(state, dt, end_time, start_time);
133 } else {
134 result = integrate_with_simple_timeout(state, dt, end_time, start_time);
135 }
136
137 } catch (const std::exception& e) {
138 result.completed = false;
139 result.error_message = "Integration failed: " + std::string(e.what());
140 }
141
142 const auto end_time_clock = std::chrono::high_resolution_clock::now();
143 result.elapsed_time = std::chrono::duration_cast<std::chrono::milliseconds>(
144 end_time_clock - start_time);
145
146 // Handle timeout according to configuration
147 if (!result.completed && config_.throw_on_timeout && result.is_timeout()) {
148 throw std::runtime_error(result.error_message);
149 }
150
151 return result;
152 }

184 TimeoutResult integrate_with_simple_timeout(
185 typename IntegratorDecorator<S, T>::state_type& state,
186 T dt, T end_time,
187 std::chrono::high_resolution_clock::time_point start_time) {
188
189 TimeoutResult result;
190
191 auto future = std::async(std::launch::async, [this, &state, dt, end_time]() {
192 this->wrapped_integrator_->integrate(state, dt, end_time);
193 });
194
195 if (future.wait_for(config_.timeout_duration) == std::future_status::timeout) {
196 result.completed = false;
197 result.error_message = "Integration timed out after " +
198 std::to_string(config_.timeout_duration.count()) + "ms";
199 } else {
200 future.get(); // May throw if integration failed
201 result.completed = true;
202 result.final_time = this->current_time();
203 }
204
205 return result;
206 }

211 TimeoutResult integrate_with_progress_monitoring(
212 typename IntegratorDecorator<S, T>::state_type& state,
213 T dt, T end_time,
214 std::chrono::high_resolution_clock::time_point start_time) {
215
216 TimeoutResult result;
217
218 auto future = std::async(std::launch::async, [this, &state, dt, end_time]() {
219 this->wrapped_integrator_->integrate(state, dt, end_time);
220 });
221
222 auto last_progress_check = start_time;
223 auto timeout_deadline = start_time + config_.timeout_duration;
224
225 // Monitor progress until completion or timeout
226 while (true) {
227 auto now = std::chrono::high_resolution_clock::now();
228 auto elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(now - start_time);
229
230 // Check if integration completed
231 if (future.wait_for(std::chrono::milliseconds{1}) == std::future_status::ready) {
232 future.get(); // May throw if integration failed
233 result.completed = true;
234 result.final_time = this->current_time();
235 break;
236 }
237
238 // Check for timeout
239 if (now >= timeout_deadline) {
240 result.completed = false;
241 result.error_message = "Integration timed out after " +
242 std::to_string(config_.timeout_duration.count()) + "ms";
243 break;
244 }
245
246 // Check if time for progress callback
247 if (now - last_progress_check >= config_.progress_interval) {
248 double current_time = this->current_time();
249 bool should_continue = config_.progress_callback(
250 current_time, static_cast<double>(end_time), elapsed);
251
252 result.progress_callbacks_made++;
253 last_progress_check = now;
254
255 // Check for user cancellation
256 if (!should_continue) {
257 result.completed = false;
258 result.user_cancelled = true;
259 result.error_message = "Integration cancelled by user";
260 break;
261 }
262 }
263
264 // Brief sleep to avoid busy waiting
265 std::this_thread::sleep_for(std::chrono::milliseconds{1});
266 }
267
268 return result;
269 }
