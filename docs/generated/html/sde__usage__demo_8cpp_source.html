156class HestonModel {
157public:
158 double mu, kappa, theta, sigma, rho;
159
160 HestonModel(double mu = 0.05, double kappa = 2.0, double theta = 0.04,
161 double sigma = 0.3, double rho = -0.7)
162 : mu(mu), kappa(kappa), theta(theta), sigma(sigma), rho(rho) {}
163
164 void run_example() {
165 std::cout << "\n=== Heston Stochastic Volatility Model ===" << std::endl;
166 std::cout << "μ = " << mu << ", κ = " << kappa << ", θ = " << theta
167 << ", σ = " << sigma << ", ρ = " << rho << std::endl;
168
169 auto drift_func = [this](double /*t*/, const std::vector<double>& x, std::vector<double>& dx) {
170 double S = x[0], V = x[1];
171 dx[0] = mu * S;
172 dx[1] = kappa * (theta - V);
173 };
174
175 auto diffusion_func = [this](double /*t*/, const std::vector<double>& x, std::vector<double>& gx) {
176 double S = x[0], V = std::max(x[1], 0.0); // Ensure V ≥ 0
177 gx[0] = std::sqrt(V) * S;
178 gx[1] = sigma * std::sqrt(V);
179 };
180
181 auto problem = diffeq::sde::factory::make_sde_problem<std::vector<double>>(
182 drift_func, diffusion_func, diffeq::sde::NoiseType::GENERAL_NOISE);
183
184 auto wiener = diffeq::sde::factory::make_wiener_process<std::vector<double>>(2, 54321);
185
186 // Set correlated noise
187 auto correlated_noise_func = [this](double /*t*/, const std::vector<double>& /*x*/,
188 std::vector<double>& noise_term, const std::vector<double>& dW) {
189 // Apply correlation: dW2_corr = ρ*dW1 + √(1-ρ²)*dW2
190 double dW1 = dW[0];
191 double dW2_corr = rho * dW1 + std::sqrt(1 - rho*rho) * dW[1];
192
193 noise_term[0] *= dW1;
194 noise_term[1] *= dW2_corr;
195 };
196
197 problem->set_noise_function(correlated_noise_func);
198
199 std::vector<double> x = {100.0, 0.04}; // Initial [S, V]
200 double dt = 0.01;
201 double T = 1.0;
202 int steps = static_cast<int>(T / dt);
203
204 // Use high-order SDE integrator for better accuracy
205 diffeq::SOSRAIntegrator<std::vector<double>> integrator(problem, wiener);
206 integrator.set_time(0.0);
207
208 std::cout << "Initial state: S = " << x[0] << ", V = " << x[1] << std::endl;
209
210 // Integrate Heston model
211 for (int i = 0; i < steps; ++i) {
212 integrator.step(x, dt);
213
214 // Output every 25 steps (quarterly)
215 if (i % 25 == 0) {
216 std::cout << "t = " << (i * dt) << ": S = " << x[0] << ", V = " << x[1] << std::endl;
217 }
218 }
219
220 std::cout << "Final state: S = " << x[0] << ", V = " << x[1] << std::endl;
221 }

240class NoisyOscillator {
241public:
242 double omega_n, zeta, sigma;
243 std::function<double(double, double, double)> control_law;
244
245 NoisyOscillator(double omega_n = 1.0, double zeta = 0.1, double sigma = 0.1)
246 : omega_n(omega_n), zeta(zeta), sigma(sigma) {
247 // Simple PD control law
248 control_law = [](double t, double x, double xdot) {
249 double target = std::sin(t); // Sinusoidal reference
250 double error = target - x;
251 double error_dot = std::cos(t) - xdot; // Derivative of reference
252 return 2.0 * error + 1.0 * error_dot; // PD gains
253 };
254 }
255
256 void run_control_example() {
257 std::cout << "\n=== Noisy Oscillator with Stochastic Control ===" << std::endl;
258 std::cout << "ωₙ = " << omega_n << ", ζ = " << zeta << ", σ = " << sigma << std::endl;
259
260 auto drift_func = [this](double t, const std::vector<double>& state, std::vector<double>& dstate) {
261 double x = state[0], xdot = state[1];
262 double u = control_law(t, x, xdot);
263
264 dstate[0] = xdot;
265 dstate[1] = -omega_n * omega_n * x - 2 * zeta * omega_n * xdot + u;
266 };
267
268 auto diffusion_func = [this](double t, const std::vector<double>& state, std::vector<double>& gstate) {
269 gstate[0] = 0.0; // No noise in position
270 gstate[1] = sigma; // Noise in velocity (acceleration)
271 };
272
273 auto problem = diffeq::sde::factory::make_sde_problem<std::vector<double>>(
274 drift_func, diffusion_func, diffeq::sde::NoiseType::DIAGONAL_NOISE);
275
276 auto wiener = diffeq::sde::factory::make_wiener_process<std::vector<double>>(1, 67890);
277
278 // For this problem, diffusion g = [0, σ], so derivative g' = [0, 0]
279 auto diffusion_derivative = [this](double /*t*/, const std::vector<double>& /*state*/, std::vector<double>& dgd_state) {
280 dgd_state[0] = 0.0; // d/dx(0) = 0
281 dgd_state[1] = 0.0; // d/dx(σ) = 0 since σ is constant
282 };
283
284 diffeq::MilsteinIntegrator<std::vector<double>> integrator(problem, diffusion_derivative, wiener);
285
286 std::vector<double> state = {0.0, 0.0}; // Initial [x, xdot]
287 double dt = 0.01;
288 double T = 10.0;
289 int steps = static_cast<int>(T / dt);
290
291 integrator.set_time(0.0);
292
293 std::cout << "Simulating controlled oscillator with noise..." << std::endl;
294
295 // Track performance metrics
296 double total_error = 0.0;
297 double max_error = 0.0;
298
299 for (int i = 0; i < steps; ++i) {
300 double t = i * dt;
301 double target = std::sin(t);
302 double error = std::abs(target - state[0]);
303
304 total_error += error;
305 max_error = std::max(max_error, error);
306
307 integrator.step(state, dt);
308
309 // Output every 100 steps
310 if (i % 100 == 0) {
311 std::cout << "t = " << t << ": x = " << state[0] << ", target = " << target
312 << ", error = " << error << std::endl;
313 }
314 }
315
316 double avg_error = total_error / steps;
317 std::cout << "Control performance:" << std::endl;
318 std::cout << " Average error: " << avg_error << std::endl;
319 std::cout << " Maximum error: " << max_error << std::endl;
320 std::cout << " Final state: x = " << state[0] << ", xdot = " << state[1] << std::endl;
321 }

341public:
342 double alpha, beta, gamma, delta, sigma1, sigma2;
343
344 StochasticLotkaVolterra(double alpha = 1.0, double beta = 1.0, double gamma = 1.0,
345 double delta = 1.0, double sigma1 = 0.1, double sigma2 = 0.1)
346 : alpha(alpha), beta(beta), gamma(gamma), delta(delta), sigma1(sigma1), sigma2(sigma2) {}
347
348 void run_ecosystem_simulation() {
349 std::cout << "\n=== Stochastic Lotka-Volterra Ecosystem Model ===" << std::endl;
350 std::cout << "α = " << alpha << ", β = " << beta << ", γ = " << gamma
351 << ", δ = " << delta << std::endl;
352 std::cout << "σ₁ = " << sigma1 << ", σ₂ = " << sigma2 << std::endl;
353
354 auto drift_func = [this](double t, const std::vector<double>& pop, std::vector<double>& dpop) {
355 double x = pop[0], y = pop[1]; // prey, predator
356 dpop[0] = alpha * x - beta * x * y;
357 dpop[1] = gamma * x * y - delta * y;
358 };
359
360 auto diffusion_func = [this](double t, const std::vector<double>& pop, std::vector<double>& gpop) {
361 double x = pop[0], y = pop[1];
362 gpop[0] = sigma1 * x; // Multiplicative noise for prey
363 gpop[1] = sigma2 * y; // Multiplicative noise for predator
364 };
365
366 auto problem = diffeq::sde::factory::make_sde_problem<std::vector<double>>(
367 drift_func, diffusion_func, diffeq::sde::NoiseType::DIAGONAL_NOISE);
368
369 auto wiener = diffeq::sde::factory::make_wiener_process<std::vector<double>>(2, 11111);
370 diffeq::SRA1Integrator<std::vector<double>> integrator(problem, wiener);
371
372 std::vector<double> population = {2.0, 1.0}; // Initial [prey, predator]
373 double dt = 0.01;
374 double T = 20.0;
375 int steps = static_cast<int>(T / dt);
376
377 integrator.set_time(0.0);
378
379 std::cout << "Initial populations: prey = " << population[0] << ", predator = " << population[1] << std::endl;
380
381 // Track population dynamics
382 double min_prey = population[0], max_prey = population[0];
383 double min_pred = population[1], max_pred = population[1];
384
385 for (int i = 0; i < steps; ++i) {
386 integrator.step(population, dt);
387
388 // Update min/max
389 min_prey = std::min(min_prey, population[0]);
390 max_prey = std::max(max_prey, population[0]);
391 min_pred = std::min(min_pred, population[1]);
392 max_pred = std::max(max_pred, population[1]);
393
394 // Output every 500 steps
395 if (i % 500 == 0) {
396 double t = i * dt;
397 std::cout << "t = " << t << ": prey = " << population[0]
398 << ", predator = " << population[1] << std::endl;
399 }
400 }
401
402 std::cout << "Final populations: prey = " << population[0] << ", predator = " << population[1] << std::endl;
403 std::cout << "Population ranges:" << std::endl;
404 std::cout << " Prey: [" << min_prey << ", " << max_prey << "]" << std::endl;
405 std::cout << " Predator: [" << min_pred << ", " << max_pred << "]" << std::endl;
406 }
