86template<typename T>
87class CubicSplineInterpolator {
88private:
89 std::vector<T> times_;
90 std::vector<std::vector<typename T::value_type>> states_;
91 std::vector<std::vector<typename T::value_type>> derivatives_;
92 bool computed_{false};
93
94public:
95 void set_data(const std::vector<T>& times, const std::vector<std::vector<typename T::value_type>>& states) {
96 times_ = times;
97 states_ = states;
98 computed_ = false;
99 compute_derivatives();
100 }
101
102 std::vector<typename T::value_type> interpolate(T t) {
103 if (!computed_) {
104 throw std::runtime_error("Spline not computed");
105 }
106
107 if (times_.empty()) {
108 throw std::runtime_error("No data for interpolation");
109 }
110
111 // Find the interval containing t
112 auto it = std::lower_bound(times_.begin(), times_.end(), t);
113
114 if (it == times_.begin()) {
115 return states_[0]; // Extrapolate to first point
116 }
117
118 if (it == times_.end()) {
119 return states_.back(); // Extrapolate to last point
120 }
121
122 size_t idx = std::distance(times_.begin(), it) - 1;
123 T h = times_[idx + 1] - times_[idx];
124 T a = (times_[idx + 1] - t) / h;
125 T b = (t - times_[idx]) / h;
126
127 std::vector<typename T::value_type> result(states_[idx].size());
128
129 for (size_t i = 0; i < result.size(); ++i) {
130 result[i] = a * states_[idx][i] + b * states_[idx + 1][i] +
131 ((a * a * a - a) * derivatives_[idx][i] + (b * b * b - b) * derivatives_[idx + 1][i]) * (h * h) / 6.0;
132 }
133
134 return result;
135 }
136
137private:
138 void compute_derivatives() {
139 size_t n = times_.size();
140 if (n < 2) return;
141
142 derivatives_.resize(n);
143 for (size_t i = 0; i < n; ++i) {
144 derivatives_[i].resize(states_[i].size());
145 }
146
147 if (n == 2) {
148 // Linear case
149 for (size_t j = 0; j < states_[0].size(); ++j) {
150 derivatives_[0][j] = derivatives_[1][j] = 0.0;
151 }
152 computed_ = true;
153 return;
154 }
155
156 // Tridiagonal system solution for cubic spline
157 std::vector<typename T::value_type> a(n), b(n), c(n);
158
159 for (size_t j = 0; j < states_[0].size(); ++j) {
160 std::vector<typename T::value_type> d(n);
161
162 // Set up tridiagonal system
163 for (size_t i = 1; i < n - 1; ++i) {
164 T h1 = times_[i] - times_[i - 1];
165 T h2 = times_[i + 1] - times_[i];
166
167 a[i] = h1;
168 b[i] = 2.0 * (h1 + h2);
169 c[i] = h2;
170 d[i] = 6.0 * ((states_[i + 1][j] - states_[i][j]) / h2 - (states_[i][j] - states_[i - 1][j]) / h1);
171 }
172
173 // Natural boundary conditions
174 b[0] = b[n - 1] = 1.0;
175 c[0] = a[n - 1] = 0.0;
176 d[0] = d[n - 1] = 0.0;
177
178 // Solve tridiagonal system
179 solve_tridiagonal(a, b, c, d);
180
181 for (size_t i = 0; i < n; ++i) {
182 derivatives_[i][j] = d[i];
183 }
184 }
185
186 computed_ = true;
187 }
188
189 void solve_tridiagonal(std::vector<typename T::value_type>& a,
190 std::vector<typename T::value_type>& b,
191 std::vector<typename T::value_type>& c,
192 std::vector<typename T::value_type>& d) {
193 size_t n = b.size();
194
195 // Forward elimination
196 for (size_t i = 1; i < n; ++i) {
197 typename T::value_type m = a[i] / b[i - 1];
198 b[i] = b[i] - m * c[i - 1];
199 d[i] = d[i] - m * d[i - 1];
200 }
201
202 // Back substitution
203 d[n - 1] = d[n - 1] / b[n - 1];
204 for (int i = n - 2; i >= 0; --i) {
205 d[i] = (d[i] - c[i] * d[i + 1]) / b[i];
206 }
207 }
208};

285 S interpolate_at(T t) {
286 std::lock_guard<std::mutex> lock(history_mutex_);
287
288 auto start_time = std::chrono::high_resolution_clock::now();
289
290 if (state_history_.empty()) {
291 throw std::runtime_error("No history available for interpolation");
292 }
293
294 // Check bounds
295 auto bounds = get_time_bounds();
296 if (t < bounds.first || t > bounds.second) {
297 if (!config_.allow_extrapolation) {
298 stats_.out_of_bounds_queries++;
299 throw std::runtime_error("Time " + std::to_string(t) + " is outside interpolation bounds [" +
300 std::to_string(bounds.first) + ", " + std::to_string(bounds.second) + "]");
301 }
302
303 // Check extrapolation warning threshold
304 T range = bounds.second - bounds.first;
305 if (std::abs(t - bounds.first) > config_.extrapolation_warning_threshold * range ||
306 std::abs(t - bounds.second) > config_.extrapolation_warning_threshold * range) {
307 stats_.extrapolation_warnings++;
308 }
309 }
310
311 S result = perform_interpolation(t);
312
313 auto end_time = std::chrono::high_resolution_clock::now();
314 double duration_ns = std::chrono::duration_cast<std::chrono::nanoseconds>(end_time - start_time).count();
315 stats_.update_interpolation_time(duration_ns);
316
317 last_query_time_ = t;
318 return result;
319 }

344 std::pair<std::vector<T>, std::vector<S>> get_dense_output(T start_time, T end_time, size_t num_points) {
345 if (num_points < 2) {

347 }

449 S linear_interpolation(T t) {
450 auto it = state_history_.lower_bound(t);
451
452 if (it == state_history_.begin()) {
453 return it->second;
454 }
455
456 if (it == state_history_.end()) {
457 return state_history_.rbegin()->second;
458 }
459
460 auto prev_it = std::prev(it);
461
462 T t1 = prev_it->first;
463 T t2 = it->first;
464 const S& s1 = prev_it->second;
465 const S& s2 = it->second;
466
467 T alpha = (t - t1) / (t2 - t1);
468
469 S result = s1;
470 for (size_t i = 0; i < result.size(); ++i) {
471 result[i] = (1 - alpha) * s1[i] + alpha * s2[i];
472 }
473
474 return result;
475 }

480 S cubic_spline_interpolation(T t) {
481 // Prepare data for spline interpolator
482 std::vector<T> times;
483 std::vector<std::vector<typename S::value_type>> states;
484
485 times.reserve(state_history_.size());
486 states.reserve(state_history_.size());
487
488 for (const auto& [time, state] : state_history_) {
489 times.push_back(time);
490 std::vector<typename S::value_type> state_vec(state.begin(), state.end());
491 states.push_back(std::move(state_vec));
492 }
493
494 spline_interpolator_->set_data(times, states);
495 auto result_vec = spline_interpolator_->interpolate(t);
496
497 // Convert back to state type
498 S result;
499 if constexpr (std::is_same_v<S, std::vector<typename S::value_type>>) {
500 result = result_vec;
501 } else {
502 std::copy(result_vec.begin(), result_vec.end(), result.begin());
503 }
504
505 return result;
506 }

527 void compress_history() {
528 if (state_history_.size() <= config_.compression_threshold) {
529 return;
530 }
531
532 // Simple compression: remove every other point if error is small
533 auto it = state_history_.begin();
534 while (it != state_history_.end() && state_history_.size() > config_.compression_threshold / 2) {
535 auto next_it = std::next(it);
536 if (next_it != state_history_.end()) {
537 auto next_next_it = std::next(next_it);
538 if (next_next_it != state_history_.end()) {
539 // Check if middle point can be removed
540 if (is_point_redundant(*it, *next_it, *next_next_it)) {
541 it = state_history_.erase(next_it);
542 continue;
543 }
544 }
545 }
546 ++it;
547 }
548
549 stats_.history_compressions++;
550 history_compressed_ = true;
551 }
552
556 bool is_point_redundant(const std::pair<T, S>& p1, const std::pair<T, S>& p2, const std::pair<T, S>& p3) {
557 // Simple linear interpolation error check
558 T alpha = (p2.first - p1.first) / (p3.first - p1.first);
559
560 for (size_t i = 0; i < p2.second.size(); ++i) {
561 double interpolated = (1 - alpha) * p1.second[i] + alpha * p3.second[i];
562 double error = std::abs(interpolated - p2.second[i]);
563 if (error > config_.compression_tolerance) {
564 return false;
565 }
566 }
567
568 return true;
569 }
