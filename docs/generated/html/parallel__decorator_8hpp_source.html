104 template<typename StateRange>
105 void integrate_batch(StateRange&& states, T dt, T end_time) {
106 const size_t batch_size = std::ranges::size(states);
107
108 if (batch_size == 0) {
109 return; // Nothing to do
110 }
111
112 if (batch_size == 1 || config_.max_threads == 1) {
113 // Sequential processing for single state or single thread
114 for (auto& state : states) {
115 this->wrapped_integrator_->integrate(state, dt, end_time);
116 }
117 return;
118 }
119
120 // Parallel processing
121 try {
122 std::for_each(std::execution::par_unseq,
123 std::ranges::begin(states), std::ranges::end(states),
124 [this, dt, end_time](auto& state) {
125 // Create thread-local copy of integrator
126 auto local_integrator = this->create_copy();
127 local_integrator->integrate(state, dt, end_time);
128 });
129 } catch (const std::exception& e) {
130 // Fall back to sequential processing if parallel fails
131 for (auto& state : states) {
132 this->wrapped_integrator_->integrate(state, dt, end_time);
133 }
134 }

149 template<typename Generator, typename Processor>
150 auto integrate_monte_carlo(size_t num_simulations, Generator&& generator,
151 Processor&& processor, T dt, T end_time) {
152 using result_type = std::invoke_result_t<Processor, S>;
153 std::vector<result_type> results(num_simulations);
154
155 if (num_simulations == 0) {
156 return results; // Nothing to do
157 }
158
159 if (num_simulations == 1 || config_.max_threads == 1) {
160 // Sequential processing
161 for (size_t i = 0; i < num_simulations; ++i) {
162 auto state = generator(i);
163 this->wrapped_integrator_->integrate(state, dt, end_time);
164 results[i] = processor(state);
165 }
166 return results;
167 }
168
169 // Parallel processing
170 try {
171 std::for_each(std::execution::par_unseq,
172 std::views::iota(0UL, num_simulations).begin(),
173 std::views::iota(0UL, num_simulations).end(),
174 [&](size_t i) {
175 auto local_integrator = this->create_copy();
176 auto state = generator(i);
177 local_integrator->integrate(state, dt, end_time);
178 results[i] = processor(state);
179 });
180 } catch (const std::exception& e) {
181 // Fall back to sequential processing if parallel fails
182 for (size_t i = 0; i < num_simulations; ++i) {
183 auto state = generator(i);
184 this->wrapped_integrator_->integrate(state, dt, end_time);
185 results[i] = processor(state);
186 }
187 }
188
189 return results;
190 }

199 template<typename StateRange>
200 void integrate_batch_chunked(StateRange&& states, T dt, T end_time) {
201 const size_t batch_size = std::ranges::size(states);
202
203 if (batch_size <= config_.chunk_size || config_.max_threads == 1) {
204 integrate_batch(std::forward<StateRange>(states), dt, end_time);
205 return;
206 }
207
208 // Calculate optimal chunk size for load balancing
209 size_t effective_chunk_size = config_.enable_auto_chunking ?
210 calculate_optimal_chunk_size(batch_size) : config_.chunk_size;
211
212 // Process in chunks
213 auto states_begin = std::ranges::begin(states);
214 auto states_end = std::ranges::end(states);
215
216 for (auto chunk_start = states_begin; chunk_start < states_end;) {
217 auto chunk_end = chunk_start;
218 std::advance(chunk_end, std::min(effective_chunk_size,
219 static_cast<size_t>(std::distance(chunk_start, states_end))));
220
221 // Create range for this chunk and process in parallel
222 std::for_each(std::execution::par_unseq, chunk_start, chunk_end,
223 [this, dt, end_time](auto& state) {
224 auto local_integrator = this->create_copy();
225 local_integrator->integrate(state, dt, end_time);
226 });
227
228 chunk_start = chunk_end;
229 }
230 }
