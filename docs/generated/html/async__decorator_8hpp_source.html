31 void validate() const {

56struct AsyncResult {
57 bool completed{false};
58 bool cancelled{false};
59 std::chrono::milliseconds execution_time{0};
60 std::string error_message;
61 size_t monitoring_checks{0};
62
63 bool is_success() const { return completed && !cancelled && error_message.empty(); }
64 bool is_cancelled() const { return cancelled; }
65 bool is_error() const { return !completed && !cancelled && !error_message.empty(); }
66
67 std::string status_description() const {
68 if (is_success()) return "Async integration completed successfully";
69 if (is_cancelled()) return "Async integration was cancelled";
70 if (is_error()) return "Async integration failed: " + error_message;
71 return "Async integration in progress";
72 }

124 std::future<AsyncResult> integrate_async(typename IntegratorDecorator<S, T>::state_type& state,
125 T dt, T end_time) {
126 return std::async(std::launch::async, [this, &state, dt, end_time]() -> AsyncResult {
127 ++active_operations_;
128 auto operation_guard = make_scope_guard([this] { --active_operations_; });
129
130 const auto start_time = std::chrono::high_resolution_clock::now();
131 AsyncResult result;
132
133 try {
134 if (config_.enable_progress_monitoring) {
135 result = integrate_with_monitoring(state, dt, end_time, start_time);
136 } else {
137 result = integrate_simple(state, dt, end_time);
138 }
139
140 if (!cancellation_requested_.load()) {
141 result.completed = true;
142 }
143
144 } catch (const std::exception& e) {
145 result.error_message = "Async integration failed: " + std::string(e.what());
146 }
147
148 const auto end_time_clock = std::chrono::high_resolution_clock::now();
149 result.execution_time = std::chrono::duration_cast<std::chrono::milliseconds>(
150 end_time_clock - start_time);
151
152 return result;
153 });
154 }

162 std::future<AsyncResult> step_async(typename IntegratorDecorator<S, T>::state_type& state, T dt) {
163 return std::async(std::launch::async, [this, &state, dt]() -> AsyncResult {
164 ++active_operations_;
165 auto operation_guard = make_scope_guard([this] { --active_operations_; });
166
167 const auto start_time = std::chrono::high_resolution_clock::now();
168 AsyncResult result;
169

172 result.cancelled = true;

183 result.execution_time = std::chrono::duration_cast<std::chrono::milliseconds>(
184 end_time_clock - start_time);

271 AsyncResult integrate_with_monitoring(typename IntegratorDecorator<S, T>::state_type& state,
272 T dt, T end_time,
273 std::chrono::high_resolution_clock::time_point start_time) {
274 AsyncResult result;
275
276 // For monitoring, we need to do step-by-step integration
277 T current_time = this->current_time();
278 auto last_check = start_time;
279
280 while (current_time < end_time && !cancellation_requested_.load()) {
281 auto now = std::chrono::high_resolution_clock::now();
282
283 // Check if it's time for monitoring
284 if (now - last_check >= config_.monitoring_interval) {
285 result.monitoring_checks++;
286 last_check = now;
287 }
288
289 // Perform one step
290 T step_size = std::min(dt, end_time - current_time);
291 this->wrapped_integrator_->step(state, step_size);
292 current_time = this->current_time();
293 }
294
295 if (cancellation_requested_.load()) {
296 result.cancelled = true;
297 }
298
299 return result;
300 }

305 template<typename F>
306 class ScopeGuard {
307 F func_;
308 bool active_;
309 public:
310 explicit ScopeGuard(F f) : func_(std::move(f)), active_(true) {}
311 ~ScopeGuard() { if (active_) func_(); }
312 ScopeGuard(ScopeGuard&& other) noexcept
313 : func_(std::move(other.func_)), active_(other.active_) {
314 other.active_ = false;
315 }
316 ScopeGuard(const ScopeGuard&) = delete;
317 ScopeGuard& operator=(const ScopeGuard&) = delete;
318 ScopeGuard& operator=(ScopeGuard&&) = delete;
319 };

322 auto make_scope_guard(F&& func) {

324 }
