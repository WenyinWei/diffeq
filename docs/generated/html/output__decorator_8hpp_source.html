42 void validate() const {
43 if (validate_intervals) {
44 if (output_interval < min_output_interval) {
45 throw std::invalid_argument("output_interval below minimum " +
46 std::to_string(min_output_interval.count()) + "μs");
47 }
48 if (output_interval > max_output_interval) {
49 throw std::invalid_argument("output_interval exceeds maximum " +
50 std::to_string(max_output_interval.count()) + "μs");

55 throw std::invalid_argument("buffer_size must be positive");
56 }

119 , config_(std::move(config))

130 ~OutputDecorator() {
131 try {
132 flush_output();

144 void step(typename IntegratorDecorator<S, T>::state_type& state, T dt) override {
145 this->wrapped_integrator_->step(state, dt);
146 ++step_count_;
147
148 handle_output(state, this->current_time());
149 }

154 void integrate(typename IntegratorDecorator<S, T>::state_type& state, T dt, T end_time) override {
155 if (config_.mode == OutputMode::OFFLINE) {
156 // Just integrate and buffer final result
157 this->wrapped_integrator_->integrate(state, dt, end_time);
158 buffer_output(state, this->current_time(), step_count_);
159 } else {
160 // Step-by-step with online output
161 while (this->current_time() < end_time) {
162 T step_size = std::min(dt, end_time - this->current_time());
163 this->step(state, step_size);
164 }
165 }
166
167 if (config_.mode == OutputMode::OFFLINE || config_.mode == OutputMode::HYBRID) {
168 flush_output();
169 }
170 }

176 void set_output_handler(std::function<void(const S&, T, size_t)> handler) {
177 output_handler_ = std::move(handler);
178 }

184 const std::vector<std::tuple<S, T, size_t>>& get_buffer() const {

186 }

191 void clear_buffer() {
192 output_buffer_.clear();
193 stats_.buffered_outputs = 0;
194 }

199 void flush_output() {
200 if (output_handler_) {
201 auto start_time = std::chrono::high_resolution_clock::now();
202
203 for (const auto& [state, time, step] : output_buffer_) {
204 output_handler_(state, time, step);
205 stats_.total_outputs++;
206 }
207
208 auto end_time = std::chrono::high_resolution_clock::now();
209 stats_.total_output_time += std::chrono::duration_cast<std::chrono::milliseconds>(
210 end_time - start_time);
211 }
212
213 if (config_.enable_file_output && output_file_ && output_file_->is_open()) {
214 write_buffer_to_file();
215 }
216 }

243 void update_config(OutputConfig new_config) {

252

258private:
262 void handle_output(const S& state, T time) {
263 auto now = std::chrono::steady_clock::now();
264
265 if (config_.mode == OutputMode::ONLINE || config_.mode == OutputMode::HYBRID) {
266 if (now - last_output_ >= config_.output_interval) {
267 if (output_handler_) {
268 auto start_time = std::chrono::high_resolution_clock::now();
269 output_handler_(state, time, step_count_);
270 auto end_time = std::chrono::high_resolution_clock::now();
271
272 stats_.total_output_time += std::chrono::duration_cast<std::chrono::milliseconds>(
273 end_time - start_time);
274 stats_.total_outputs++;
275 stats_.online_outputs++;
276 }
277 last_output_ = now;
278 }
279 }
280
281 if (config_.mode == OutputMode::OFFLINE || config_.mode == OutputMode::HYBRID) {
282 buffer_output(state, time, step_count_);
283 }
284 }
285
289 void buffer_output(const S& state, T time, size_t step) {
290 if (output_buffer_.size() >= config_.buffer_size) {
291 output_buffer_.erase(output_buffer_.begin());
292 stats_.buffer_overflows++;
293 }
294 output_buffer_.emplace_back(state, time, step);
295 stats_.buffered_outputs++;
296 }
297
301 void initialize_file_output() {
302 if (config_.enable_file_output && !config_.output_filename.empty()) {
303 output_file_ = std::make_unique<std::ofstream>();
304
305 std::ios_base::openmode mode = std::ios_base::out;
306 if (config_.append_to_file) {
307 mode |= std::ios_base::app;
308 }
309
310 output_file_->open(config_.output_filename, mode);
311
312 if (!output_file_->is_open()) {
313 throw std::runtime_error("Failed to open output file: " + config_.output_filename);
314 }
315
316 // Write header if creating new file
317 if (!config_.append_to_file) {
318 *output_file_ << "# DiffEq Integration Output\n";
319 *output_file_ << "# Time, State, Step\n";
320 }
321 }
322 }
323
327 void write_buffer_to_file() {
328 if (!output_file_ || !output_file_->is_open()) {
329 return;
330 }
331
332 for (const auto& [state, time, step] : output_buffer_) {
333 *output_file_ << time << ", ";
334
335 // Write state (assuming it's iterable)
336 bool first = true;
337 for (const auto& component : state) {
338 if (!first) *output_file_ << " ";
339 *output_file_ << component;
340 first = false;
341 }
342
343 *output_file_ << ", " << step << "\n";
344 stats_.file_writes++;
345 }
346
347 output_file_->flush();
348 }
349};
