214template<typename T>
215class SharedMemoryChannel : public IPCChannel<T> {
216private:
217 std::string name_;
218 size_t buffer_size_;
219 void* memory_ptr_{nullptr};
220
221#ifdef _WIN32
222 HANDLE file_mapping_{nullptr};
223#else
224 int fd_{-1};
225#endif
226
227 std::mutex* mutex_{nullptr};
228 std::condition_variable* condition_{nullptr};
229 bool initialized_{false};
230
231public:
232 explicit SharedMemoryChannel(const std::string& name, size_t buffer_size)
233 : name_(name), buffer_size_(buffer_size) {}
234
235 ~SharedMemoryChannel() {
236 cleanup();
237 }
238
239 bool initialize() override {
240 if (initialized_) return true;
241
242#ifdef _WIN32
243 file_mapping_ = CreateFileMapping(
244 INVALID_HANDLE_VALUE,
245 nullptr,
246 PAGE_READWRITE,
247 0,
248 buffer_size_,
249 name_.c_str()
250 );
251
252 if (file_mapping_ == nullptr) {
253 return false;
254 }
255
256 memory_ptr_ = MapViewOfFile(
257 file_mapping_,
258 FILE_MAP_ALL_ACCESS,
259 0,
260 0,
261 buffer_size_
262 );
263
264 if (memory_ptr_ == nullptr) {
265 CloseHandle(file_mapping_);
266 return false;
267 }
268#else
269 fd_ = shm_open(name_.c_str(), O_CREAT | O_RDWR, 0666);
270 if (fd_ == -1) {
271 return false;
272 }
273
274 if (ftruncate(fd_, buffer_size_) == -1) {
275 close(fd_);
276 return false;
277 }
278
279 memory_ptr_ = mmap(nullptr, buffer_size_, PROT_READ | PROT_WRITE, MAP_SHARED, fd_, 0);
280 if (memory_ptr_ == MAP_FAILED) {
281 close(fd_);
282 return false;
283 }
284#endif
285
286 // Initialize synchronization primitives in shared memory
287 mutex_ = new(memory_ptr_) std::mutex();
288 condition_ = new(static_cast<char*>(memory_ptr_) + sizeof(std::mutex)) std::condition_variable();
289
290 initialized_ = true;
291 return true;
292 }
293
294 void cleanup() override {
295 if (!initialized_) return;
296
297#ifdef _WIN32
298 if (memory_ptr_) {
299 UnmapViewOfFile(memory_ptr_);
300 memory_ptr_ = nullptr;
301 }
302 if (file_mapping_) {
303 CloseHandle(file_mapping_);
304 file_mapping_ = nullptr;
305 }
306#else
307 if (memory_ptr_ != MAP_FAILED) {
308 munmap(memory_ptr_, buffer_size_);
309 memory_ptr_ = nullptr;
310 }
311 if (fd_ != -1) {
312 close(fd_);
313 shm_unlink(name_.c_str());
314 fd_ = -1;
315 }
316#endif
317
318 initialized_ = false;
319 }
320
321 bool send_message(const IPCMessage<T>& message) override {
322 if (!initialized_) return false;
323
324 std::lock_guard<std::mutex> lock(*mutex_);
325
326 // Write message to shared memory
327 char* data_ptr = static_cast<char*>(memory_ptr_) + sizeof(std::mutex) + sizeof(std::condition_variable);
328 size_t message_size = sizeof(IPCMessage<T>) + message.data.size();
329
330 if (message_size > buffer_size_ - sizeof(std::mutex) - sizeof(std::condition_variable)) {
331 return false;
332 }
333
334 std::memcpy(data_ptr, &message, sizeof(IPCMessage<T>));
335 std::memcpy(data_ptr + sizeof(IPCMessage<T>), message.data.data(), message.data.size());
336
337 condition_->notify_one();
338 return true;
339 }
340
341 bool receive_message(IPCMessage<T>& message) override {
342 if (!initialized_) return false;
343
344 std::unique_lock<std::mutex> lock(*mutex_);
345
346 // For simplicity, we'll use a timeout-based approach
347 if (!condition_->wait_for(lock, std::chrono::milliseconds(100))) {
348 return false;
349 }
350
351 // Read message from shared memory
352 char* data_ptr = static_cast<char*>(memory_ptr_) + sizeof(std::mutex) + sizeof(std::condition_variable);
353
354 std::memcpy(&message, data_ptr, sizeof(IPCMessage<T>));
355 message.data.resize(message.message_size);
356 std::memcpy(message.data.data(), data_ptr + sizeof(IPCMessage<T>), message.message_size);
357
358 return true;
359 }
360
361 bool is_connected() const override {
362 return initialized_;
363 }
364
365 std::string get_status() const override {
366 return initialized_ ? "Connected" : "Disconnected";
367 }
368};

373template<typename T>
374class NamedPipeChannel : public IPCChannel<T> {
375private:
376 std::string name_;
377 std::string pipe_path_;
378
379#ifdef _WIN32
380 HANDLE pipe_handle_{INVALID_HANDLE_VALUE};
381#else
382 int pipe_fd_{-1};
383#endif
384
385 bool initialized_{false};
386 bool is_server_{false};
387
388public:
389 explicit NamedPipeChannel(const std::string& name, bool is_server = false)
390 : name_(name), is_server_(is_server) {
391
392#ifdef _WIN32
393 pipe_path_ = "\\\\.\\pipe\\" + name_;
394#else
395 pipe_path_ = "/tmp/" + name_;
396#endif
397 }
398
399 ~NamedPipeChannel() {
400 cleanup();
401 }
402
403 bool initialize() override {
404 if (initialized_) return true;
405
406#ifdef _WIN32
407 if (is_server_) {
408 pipe_handle_ = CreateNamedPipe(
409 pipe_path_.c_str(),
410 PIPE_ACCESS_DUPLEX,
411 PIPE_TYPE_BYTE | PIPE_READMODE_BYTE | PIPE_WAIT,
412 1,
413 4096,
414 4096,
415 0,
416 nullptr
417 );
418
419 if (pipe_handle_ == INVALID_HANDLE_VALUE) {
420 return false;
421 }
422
423 if (!ConnectNamedPipe(pipe_handle_, nullptr)) {
424 if (GetLastError() != ERROR_PIPE_CONNECTED) {
425 CloseHandle(pipe_handle_);
426 return false;
427 }
428 }
429 } else {
430 pipe_handle_ = CreateFile(
431 pipe_path_.c_str(),
432 GENERIC_READ | GENERIC_WRITE,
433 0,
434 nullptr,
435 OPEN_EXISTING,
436 0,
437 nullptr
438 );
439
440 if (pipe_handle_ == INVALID_HANDLE_VALUE) {
441 return false;
442 }
443 }
444#else
445 if (is_server_) {
446 if (mkfifo(pipe_path_.c_str(), 0666) == -1 && errno != EEXIST) {
447 return false;
448 }
449 }
450
451 pipe_fd_ = open(pipe_path_.c_str(), O_RDWR | O_NONBLOCK);
452 if (pipe_fd_ == -1) {
453 return false;
454 }
455#endif
456
457 initialized_ = true;
458 return true;
459 }
460
461 void cleanup() override {
462 if (!initialized_) return;
463
464#ifdef _WIN32
465 if (pipe_handle_ != INVALID_HANDLE_VALUE) {
466 CloseHandle(pipe_handle_);
467 pipe_handle_ = INVALID_HANDLE_VALUE;
468 }
469#else
470 if (pipe_fd_ != -1) {
471 close(pipe_fd_);
472 pipe_fd_ = -1;
473 }
474 if (is_server_) {
475 unlink(pipe_path_.c_str());
476 }
477#endif
478
479 initialized_ = false;
480 }
481
482 bool send_message(const IPCMessage<T>& message) override {
483 if (!initialized_) return false;
484
485 // Serialize message
486 std::vector<uint8_t> serialized_data;
487 serialized_data.resize(sizeof(IPCMessage<T>) + message.data.size());
488
489 std::memcpy(serialized_data.data(), &message, sizeof(IPCMessage<T>));
490 std::memcpy(serialized_data.data() + sizeof(IPCMessage<T>), message.data.data(), message.data.size());
491
492#ifdef _WIN32
493 DWORD bytes_written;
494 return WriteFile(pipe_handle_, serialized_data.data(), serialized_data.size(), &bytes_written, nullptr) &&
495 bytes_written == serialized_data.size();
496#else
497 ssize_t bytes_written = write(pipe_fd_, serialized_data.data(), serialized_data.size());
498 return bytes_written == static_cast<ssize_t>(serialized_data.size());
499#endif
500 }
501
502 bool receive_message(IPCMessage<T>& message) override {
503 if (!initialized_) return false;
504
505 // First, read the message header
506 IPCMessage<T> header;
507
508#ifdef _WIN32
509 DWORD bytes_read;
510 if (!ReadFile(pipe_handle_, &header, sizeof(IPCMessage<T>), &bytes_read, nullptr) ||
511 bytes_read != sizeof(IPCMessage<T>)) {
512 return false;
513 }
514#else
515 ssize_t bytes_read = read(pipe_fd_, &header, sizeof(IPCMessage<T>));
516 if (bytes_read != sizeof(IPCMessage<T>)) {
517 return false;
518 }
519#endif
520
521 // Then read the data
522 message = header;
523 message.data.resize(header.message_size);
524
525#ifdef _WIN32
526 if (!ReadFile(pipe_handle_, message.data.data(), header.message_size, &bytes_read, nullptr) ||
527 bytes_read != header.message_size) {
528 return false;
529 }
530#else
531 bytes_read = read(pipe_fd_, message.data.data(), header.message_size);
532 if (bytes_read != static_cast<ssize_t>(header.message_size)) {
533 return false;
534 }
535#endif
536
537 return true;
538 }
539
540 bool is_connected() const override {
541 return initialized_;
542 }
543
544 std::string get_status() const override {
545 return initialized_ ? "Connected" : "Disconnected";
546 }

566class InterprocessDecorator : public IntegratorDecorator<S, T> {
567private:
568 InterprocessConfig config_;
569 std::unique_ptr<IPCChannel<T>> channel_;
570 std::atomic<uint32_t> sequence_number_{0};
571 IPCStats stats_;
572
573 // Threading for async operations
574 std::thread communication_thread_;
575 std::atomic<bool> running_{false};
576 std::mutex message_queue_mutex_;
577 std::condition_variable message_queue_cv_;
578 std::queue<IPCMessage<T>> outgoing_messages_;
579 std::queue<IPCMessage<T>> incoming_messages_;
580
581 // Callbacks for received data
582 std::function<void(const S&, T)> receive_callback_;
583
584 // SDE synchronization
585 std::mutex sde_sync_mutex_;
586 std::condition_variable sde_sync_cv_;
587 std::atomic<bool> noise_data_available_{false};
588 S pending_noise_data_;
589
590public:
597 explicit InterprocessDecorator(std::unique_ptr<AbstractIntegrator<S, T>> integrator,
598 InterprocessConfig config = {})
599 : IntegratorDecorator<S, T>(std::move(integrator)), config_(std::move(config)) {
600
601 config_.validate();
602 initialize_channel();
603
604 if (config_.direction == IPCDirection::CONSUMER || config_.direction == IPCDirection::BIDIRECTIONAL) {
605 start_communication_thread();
606 }
607 }
608
612 ~InterprocessDecorator() {
613 stop_communication_thread();
614 if (channel_) {
615 channel_->cleanup();
616 }
617 }
618
622 void step(typename IntegratorDecorator<S, T>::state_type& state, T dt) override {
623 // Handle SDE synchronization if needed
624 if (config_.direction == IPCDirection::CONSUMER) {
625 wait_for_noise_data();
626 }
627
628 // Send current state if producer
629 if (config_.direction == IPCDirection::PRODUCER || config_.direction == IPCDirection::BIDIRECTIONAL) {
630 send_state(state, this->current_time());
631 }
632
633 // Perform integration step
634 this->wrapped_integrator_->step(state, dt);
635
636 // Handle received data if consumer
637 if (config_.direction == IPCDirection::CONSUMER || config_.direction == IPCDirection::BIDIRECTIONAL) {
638 process_incoming_messages(state);
639 }
640 }
641
645 void integrate(typename IntegratorDecorator<S, T>::state_type& state, T dt, T end_time) override {
646 // Send initial state
647 if (config_.direction == IPCDirection::PRODUCER || config_.direction == IPCDirection::BIDIRECTIONAL) {
648 send_state(state, this->current_time());
649 }
650
651 // Integrate with IPC handling
652 this->wrapped_integrator_->integrate(state, dt, end_time);
653
654 // Send final state
655 if (config_.direction == IPCDirection::PRODUCER || config_.direction == IPCDirection::BIDIRECTIONAL) {
656 send_state(state, this->current_time());
657 }
658 }
659
664 void set_receive_callback(std::function<void(const S&, T)> callback) {
665 receive_callback_ = std::move(callback);
666 }
667
674 bool send_state(const S& state, T time) {
675 if (!channel_ || !channel_->is_connected()) {
676 return false;
677 }
678
679 auto start_time = std::chrono::high_resolution_clock::now();
680
681 IPCMessage<T> message;
682 message.sequence_number = sequence_number_++;
683 message.timestamp = time;
684 message.serialize_state(state);
685
686 bool success = false;
687 size_t retries = 0;
688
689 while (!success && retries < config_.max_retries) {
690 success = channel_->send_message(message);
691 if (!success) {
692 retries++;
693 std::this_thread::sleep_for(std::chrono::milliseconds(10));
694 }
695 }
696
697 auto end_time = std::chrono::high_resolution_clock::now();
698 auto duration = std::chrono::duration_cast<std::chrono::milliseconds>(end_time - start_time);
699
700 if (success) {
701 stats_.messages_sent++;
702 stats_.bytes_sent += message.message_size;
703 stats_.total_send_time += duration;
704 } else {
705 stats_.send_failures++;
706 }
707
708 return success;
709 }
710
714 const IPCStats& get_statistics() const {
715 return stats_;
716 }

723 }
724

736 return channel_ ? channel_->get_status() : "Not initialized";

742 InterprocessConfig& config() { return config_; }

745private:
749 void initialize_channel() {
750 switch (config_.method) {
751 case IPCMethod::SHARED_MEMORY:
752 channel_ = std::make_unique<SharedMemoryChannel<T>>(config_.channel_name, config_.buffer_size);
753 break;
754 case IPCMethod::NAMED_PIPES:
755 channel_ = std::make_unique<NamedPipeChannel<T>>(config_.channel_name,
756 config_.direction == IPCDirection::PRODUCER);
757 break;
758 case IPCMethod::MEMORY_MAPPED_FILE:
759 // TODO: Implement memory-mapped file channel
760 throw std::runtime_error("Memory-mapped file IPC not yet implemented");
761 case IPCMethod::TCP_SOCKET:
762 // TODO: Implement TCP socket channel
763 throw std::runtime_error("TCP socket IPC not yet implemented");
764 case IPCMethod::UDP_SOCKET:
765 // TODO: Implement UDP socket channel
766 throw std::runtime_error("UDP socket IPC not yet implemented");
767 default:
768 throw std::runtime_error("Unknown IPC method");
769 }
770
771 if (!channel_->initialize()) {
772 throw std::runtime_error("Failed to initialize IPC channel");
773 }
774 }
775
779 void start_communication_thread() {
780 running_ = true;
781 communication_thread_ = std::thread([this]() {
782 while (running_) {
783 IPCMessage<T> message;
784 if (channel_->receive_message(message)) {
785 std::lock_guard<std::mutex> lock(message_queue_mutex_);
786 incoming_messages_.push(message);
787 message_queue_cv_.notify_one();
788
789 stats_.messages_received++;
790 stats_.bytes_received += message.message_size;
791 }
792
793 std::this_thread::sleep_for(config_.polling_interval);
794 }
795 });
796 }
797
801 void stop_communication_thread() {
802 running_ = false;
803 if (communication_thread_.joinable()) {
804 communication_thread_.join();
805 }
806 }
807
811 void process_incoming_messages(S& state) {
812 std::lock_guard<std::mutex> lock(message_queue_mutex_);
813
814 while (!incoming_messages_.empty()) {
815 auto message = incoming_messages_.front();
816 incoming_messages_.pop();
817
818 // Deserialize state data
819 auto received_state = message.deserialize_state();
820
821 // Call receive callback if set
822 if (receive_callback_) {
823 receive_callback_(received_state, message.timestamp);
824 }
825
826 // For SDE synchronization, signal that noise data is available
827 if (config_.direction == IPCDirection::CONSUMER) {
828 std::lock_guard<std::mutex> sync_lock(sde_sync_mutex_);
829 pending_noise_data_ = received_state;
830 noise_data_available_ = true;
831 sde_sync_cv_.notify_one();
832 }
833 }
834 }
835
839 void wait_for_noise_data() {
840 std::unique_lock<std::mutex> lock(sde_sync_mutex_);
841
842 if (config_.sync_mode == IPCSyncMode::BLOCKING) {
843 sde_sync_cv_.wait(lock, [this] { return noise_data_available_.load(); });
844 } else if (config_.sync_mode == IPCSyncMode::TIMEOUT) {
845 sde_sync_cv_.wait_for(lock, config_.timeout, [this] { return noise_data_available_.load(); });
846 }
847
848 noise_data_available_ = false;
849 }
850};
