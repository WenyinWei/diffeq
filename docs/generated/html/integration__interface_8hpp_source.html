82 template<typename SignalDataType>
83 void register_signal_influence(
84 std::string_view signal_type,
85 InfluenceMode mode,
86 std::function<void(const SignalDataType&, state_type&, time_type)> handler,
87 double priority = 1.0) {
88
89 SignalInfluence influence{
90 .mode = mode,
91 .signal_type = std::string(signal_type),
92 .handler = [h = std::move(handler)](const std::any& signal_data, state_type& state, time_type t) {
93 try {
94 const auto& typed_data = std::any_cast<const SignalDataType&>(signal_data);
95 h(typed_data, state, t);
96 } catch (const std::bad_any_cast&) {
97 // Type mismatch - ignore silently
98 }
99 },
100 .priority = priority,
101 .is_active = true
102 };
103
104 signal_influences_.push_back(std::move(influence));
105
106 // Register with signal processor
107 signal_processor_->template register_handler<SignalDataType>(signal_type,
108 [this, signal_type_str = std::string(signal_type)](const signal::Signal<SignalDataType>& sig) {
109 handle_signal(signal_type_str, sig.data);
110 });
111 }

149 template<typename OriginalODE>
150 auto make_signal_aware_ode(OriginalODE&& original_ode) {
151 return [this, ode = std::forward<OriginalODE>(original_ode)]
152 (time_type t, const state_type& y, state_type& dydt) {
153
154 // Update current state for signal handling
155 current_state_ = y;
156 current_time_ = t;
157
158 // Compute original dynamics
159 ode(t, y, dydt);
160
161 // Apply trajectory modifiers from continuous signal influences
162 state_type modified_state = y;
163 for (auto& modifier : trajectory_modifiers_) {
164 modifier(t, modified_state, dydt);
165 }
166
167 // Handle real-time outputs
168 process_output_streams(y, t);
169 };
170 }

175 void apply_discrete_event(const std::string& signal_type, const std::any& signal_data) {
176 if (!current_state_) return;

178 for (auto& influence : signal_influences_) {
179 if (influence.signal_type == signal_type &&
180 influence.mode == InfluenceMode::DISCRETE_EVENT &&
181 influence.is_active) {

183 influence.handler(signal_data, *current_state_, current_time_);

236 void handle_signal(const std::string& signal_type, const std::any& signal_data) {
237 // Handle different influence modes
238 for (auto& influence : signal_influences_) {
239 if (influence.signal_type == signal_type && influence.is_active) {
240
241 if (influence.mode == InfluenceMode::DISCRETE_EVENT && current_state_) {
242 // Apply immediate state modification
243 influence.handler(signal_data, *current_state_, current_time_);
244
245 } else if (influence.mode == InfluenceMode::CONTINUOUS_SHIFT) {
246 // Add to trajectory modifiers for continuous influence
247 add_trajectory_modifier([influence, signal_data]
248 (time_type t, state_type& state, state_type& /* dydt */) {
249 influence.handler(signal_data, state, t);
250 });
251
252 } else if (influence.mode == InfluenceMode::PARAMETER_UPDATE) {
253 // Handle parameter updates (integrator-specific)
254 if (current_state_) {
255 influence.handler(signal_data, *current_state_, current_time_);
256 }
257
258 } else if (influence.mode == InfluenceMode::OUTPUT_TRIGGER && current_state_) {
259 // Trigger immediate output
260 for (auto& stream : output_streams_) {
261 if (stream.is_active) {
262 stream.output_func(*current_state_, current_time_);
263 }
264 }
265 }
266 }
267 }
268 }
269
270 void process_output_streams(const state_type& state, time_type t) {
271 auto now = std::chrono::steady_clock::now();
272
273 for (auto& stream : output_streams_) {
274 if (!stream.is_active) continue;
275
276 if (now - stream.last_output >= stream.interval) {
277 stream.output_func(state, t);
278 stream.last_output = now;
279 }
280 }
281 }
